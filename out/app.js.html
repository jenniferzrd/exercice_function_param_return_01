  var p = {};

  function isEmpty(p) {
    if (p === "") {
      return true;
    }
    else if (Array.isArray(p) && p.length === 0) {
      return true;
    }
    else if (typeof(p) === 'object' && Object.getOwnPropertyNames(p).length === 0) {
      return true;
    } else {
      return false;
    }

  }
  console.log("resultat exo 8");
  isEmpty();
  console.log(isEmpty(p));
  var x = !isEmpty(p);
  console.log(x);
